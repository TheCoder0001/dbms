import mysql.connector

con = mysql.connector.connect(
    host="localhost",
    user="root",
    password="",
    database="test"
)

cur = con.cursor()

# Insert
cur.execute("INSERT INTO emp VALUES(1,'Rohan',55000)")
con.commit()

# Display
cur.execute("SELECT * FROM emp")
for row in cur.fetchall():
    print(row)

# Update
cur.execute("UPDATE emp SET salary=60000 WHERE emp_no=1")
con.commit()

# Delete
cur.execute("DELETE FROM emp WHERE emp_no=1")
con.commit()

con.close()
