PHP/python/Java Implement Database navigation CRUD operations

from pymongo import MongoClient

client = MongoClient("mongodb://localhost:27017")
db = client.mydb
emp = db.Employee

# Insert
emp.insert_one({"Name":"Rohan","Company":"TCS","Salary":55000})

# Display
for e in emp.find():
    print(e)

# Update
emp.update_one({"Name":"Rohan"},{"$set":{"Salary":60000}})

# Delete
emp.delete_one({"Name":"Rohan"})
